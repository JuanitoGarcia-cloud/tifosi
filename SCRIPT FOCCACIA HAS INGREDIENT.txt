CREATE TABLE IF NOT EXISTS `tifosi`.`foccacia_has_ingredient` (
  `nom_ingredient` VARCHAR(50) NOT NULL,
  `quantite` INT(11) NULL,
  `nom_foccacia` VARCHAR(50) NOT NULL,
  PRIMARY KEY (`nom_foccacia`)
  INDEX `fk_foccacia_has_ingredient_ingredient_idx` (`nom_ingredient` ASC),
  INDEX `fk_foccacia_has_ingredient_foccacia_idx` (`nom_foccacia` ASC),
  CONSTRAINT `fk_foccacia_has_ingredient_foccacia`
    FOREIGN KEY (`nom_foccacia`)
    REFERENCES `tifosi`.`foccacia` (`nom_foccacia`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_foccacia_has_ingredient_ingredient`
    FOREIGN KEY (`nom_ingredient`)
    REFERENCES `tifosi`.`ingredient` (`nom_ingredient`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;
